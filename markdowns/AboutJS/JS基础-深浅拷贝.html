    # JS基础-深浅拷贝

    深浅拷贝都是指对一个对象或者数组进行复制。

    在react中，一个很重要的思想就是state的不可变性。这就要求我们setState之前，不应当修改state中的任何属性。这个时候，我们就需要实现深拷贝，来确保不可变性。

    # 浅拷贝：

    复制对象或者数组的引用。通过这样拷贝出来的对象，其引用类型的属性是会多个实例之间共享的。

    let a=[1,2,3];
let b=a;
b[0]=4;
console.log(a);	//4,2,3
console.log(b);	//4,2,3

    # 深拷贝

    # 1，通过ES6的解构赋值可以实现深拷贝

    let a=[1,2,3];
const [...b]=a;
b[0]=4;
console.log(a);	//1,2,3
console.log(b);	//4,2,3---------------->实现了深拷贝

    # 2，Object.assign实现的是浅拷贝

    let a=[1,2,3];
let b=Object.assign(a);
b[0]=4;
console.log(a);	//4，2，3
console.log(b);	//4，2，3

    # 3，通过JSON.parse实现深拷贝

    let a=[1,2,3];
let b=JSON.parse(JSON.stringify(a));
console.log(a);	//[1,2,3]
b[0]=4;
console.log(a);	//[1,2,3]
console.log(b);	//[4,2,3]---------------->实现了深拷贝

    # 4，自己实现一个深拷贝

    function deepClone(origin){
	if(typeof origin!=='object'){
        return origin;	//如果不是对象类型，则直接返回,typeof array也属于对象类型
    }
    var targetObj=Array.isArray(origin)?[]:{};	//判断是数组还是对象
    
    Object.keys(origin).forEach(key=>{
     	if(typeof origin[key]==='object'){	//遍历属性，如果属性为object类型，则继续深度遍历
        	targetObj[key]=deepClone(origin[key])	
        }else{
            targetObj[key]=origin[key];
        }   
    })
    return targetObj;
}

let a=[1,2,3];
let b=deepClone(a);
console.log(a);	//[1,2,3]
b[0]=4;
console.log(a);	//[1,2,3]
console.log(b);	//[4,2,3]

let studentA={name:'zhangsan',age:14,parent:{father:'jack',mother:'lily'}};
let studentB=deepClone(studentA);
studentB.parent.father='join';

console.log(studentB);	//father:join,mother:lily
console.log(studentA);	//father:jack,mother:lily

    # 5，Array.slice和Array.concat

    //我们可以使用slice和concat对数组元素进行拷贝
let a=[1,2,3];
let b=a.slice();
let c=a.concat();

console.log(b);	//[1,2,3];
console.log(c);	//[1,2,3];

b[0]=100;
console.log(a);	//[1,2,3]
console.log(b);	//[100,2,3]	----------------------->实现了深拷贝-------吗？
console.log(c);	//[1,2,3]

    WARNING

    这两个方法看上去是深拷贝，实际上仅仅对数组的第一层进行了深拷贝，到了第二层依然是直接复制了引用，所以也属于浅拷贝。

    let arrA=[1,2,3,[4,5,6]];
let arrB=arrA.slice();
arrB[3][0]=100;

console.log(arrB);	//[1,2,3,[100,5,6]]
console.log(arrA);	//[1,2,3,[100,5,6]]------------------>深拷贝失败，a也在b修改时被修改了
