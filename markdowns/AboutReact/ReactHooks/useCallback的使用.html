      # useCallback的使用

      # 1，基本使用

      useCallback返回了一个memoized的函数。通过这种方式，可以保证当组件更新时，函数不会重新创建，从而避免了子组件的刷新。

      const increment2 = useCallback(() => {
    console.log('increment2 重新执行')
    setCounter(counter+1)
}, [counter])			//当counter不变时，increment也不会发生改变。引用了increment的组件也不会更新。

      # 2，示例代码

      import React, { useState, useEffect, useCallback ,memo } from 'react'

const FAButton= memo(function(props){
  console.log('button 重新渲染',props.title);
  return <button onClick={props.increment}>Fan Button +1</button>
})

export default function CallbackHookDemo2() {

  const [show, setShow] = useState(true)
  const [counter, setCounter] = useState(0)

  const increment1 = () => {
    console.log('increment1 重新执行');
    setCounter(counter+1)
  }

  const increment2 = useCallback(() => {
    console.log('increment2 重新执行')
    setCounter(counter+1)
  }, [counter])

  return (
    <div>
      <h2>callback hook demo 02 : callback能进行的性能优化</h2>
      <h2>当前计数：{counter}</h2>
      <FAButton title='btn1' increment={increment1}></FAButton>
      <FAButton title='btn2' increment={increment2}></FAButton>
      <button onClick={e=>setShow(!show)}>SetShow</button>     
    </div>
  )
}
