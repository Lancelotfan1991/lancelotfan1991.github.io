        # useEffect的使用

        # 1，基本使用

        useEffect为函数式组件提供了一个和类组件中生命周期函数用法相近的hook。从而可以帮助我们实现诸如手动修改Dom，发送网络请求，订阅事件和取消订阅等功能。

        而且与生命周期函数不同，useEffect可以针对性地拆分不同的副作用，从而可以提高性能。并且减少了代码重复。

        useEffect(()=>{})	//------------------------------每次更新state均会触发

useEffect(()=>{},[])	//--------------------------不牵涉任何state，整个生命周期仅仅会调用一次

useEffect(()=>{},[stateA,stateB])  //---------------当stateA或者stateB改变时，都会调用

        # 2，示例代码

        import React, { useState, useEffect } from 'react'

export default function CancelSubscribe() {
  
  const [ counter, setCounter] = useState(0)
  const [ age , setAge ] = useState(1);

  //useEffect的第二个参数代表监听哪些状态的变化
  //只有相应的状态变化了,才会重新触发
  //如果监听的状态是空数组,那么就代表着,当状态变化时,不会重新触发这个副作用,除非函数式组件销毁并重新创建;
  //换句话说-----这个副作用只会执行一次
  //而如果第三个参数什么都不写,那么只要状态发生改变,那么副作用总会被触发
  //出于性能考虑,应当尽可能填好第三个参数,避免副作用一直触发
  
  useEffect(() => { 
    console.log('订阅某个事件，这个effect在组件的整个生命周期中只会触发一次')
    return () => {
      console.log('取消订阅某个事件')
    }
  }, [])

  useEffect(() => {
    console.log('触发了这个副作用,不管修改的不是该状态')
    document.title=counter;
  })

  useEffect(()=>{ 
    console.log('仅会在特定state更新时,才会触发')
    console.log(counter);
  },[counter])

  return (
    <div>
      <h2>{counter}</h2>
      <h2>{age}</h2>
      <button onClick={ e => setCounter(counter + 1)}>计数+1</button>
      <button onClick={ e => setAge(age+1)}>年龄+1</button>
    </div>
  )
}
