# Redux-Saga

# 1，引入redux-saga：

//第一步，引入
import { createStore , applyMiddleware , compose} from 'redux';
import createSageMiddleware from 'redux-saga';
import sage from './saga';//自己创建的sage.js，用于规定saga拦截的action

const composeEnhancers = window.__REDUX_DEVTOOLS_EXTENSION_COMPOSE__({trace:true}) || compose

//第二步，创建
const sagaMiddleware = createSageMiddleware();

//第三步，应用中间件
const storeEnhancer = applyMiddleware( sagaMiddleware );

//第四步，创建store
const store = createStore(reducer , composeEnhancer(storeEnhancer));

//第五步，运行saga
sagaMiddleware.run(saga);

export default store;

# 2，saga.js

import {takeEvery , put , all , takeLatest} from 'redux-saga/effects';
import axios from 'axios';

//获取saga 想要获取的 action的类型
import { FETCH_HOME_MULTIDATA } from './constants'
//获取saga 拦截后 想要执行的action
import { action1 ,action2 } from './actionCreator';

//真正的拦截操作,这是一个generator,后面带了*号
function* fetch_home_multidata(){
   const res = yield axios.get('abs/123/getdata');
   const banners = res.data.data.banner.list;
   const recommend = res.data.data.recommend.list;
   
   yield all([
       yield action1(banners);
       yield action2(recommend);
   ]);
}

//创建saga的匹配关系，将派发的事件和具体的异步操作联系起来，这也是一个generator
function* mySaga(){
   yield all([
       takeEvery( FETCH_HOME_MULTIDATA , fetch_home_multidata);
   ]) 
}

export default mySaga;
