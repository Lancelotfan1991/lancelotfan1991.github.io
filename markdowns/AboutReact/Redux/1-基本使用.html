        # Redux-1

        Redux是Javascript的状态管理容器，提供了可预测的状态管理。

        # 1，Redux的核心理念

        # -store:存储数据

        # -action：所有的数据必须通过action，派发dispatch action来更新

        const action1= {type:"ADD_FRIEND",info:{name:"lucy",age:20}}
const action2= {type:"INC_AGE",index:0}

        # -reducer：将store和action联系起来

        //reducer是一个纯函数，他接受两个参数
//因为是纯函数，所以不允许产生任何副作用
function reducer(state = initialState,action){
    switch(action.type){
        case "ACTION_1":
            return {...state,friends:[...state.friends,'JACK']}
        case "ACTION_2":
            return {...state}
    }
}

        # 2，Redux的三个原则

        • 单一数据源，只有一个store
        • state是只读的，唯一修改state的方法是触发action
        • 使用纯函数来执行修改

        # 3，Redux的基本使用-在非React环境中使用

        import redux from 'redux';

//创建初始状态
const initialState = {
	counter:0;
}

//创建reducer
const reducer = function(state = initialState , action){
    switch(action.type){
        case 'INCREMENT':
            return {...state,counter:state.counter+1}
        case 'DECREMENT':
            return {...state,counter:state.counter-1}
        case "ADD_NUMBER":
            return {...state,counter:state.counter+action.num}
        case "SUB_NUMBER":
            return {...state,counter:state.counter-action.num}
        default:
            return state
    }
}

//创建store
const store=redux.createStore(reducer);

//定义action
const action1={type:'INCREMENT'},
const action2={type:'DECREMENT'},
      action3={type:'ADD_NUMBER',num:5},
      action4={type:'SUB_NUMBER',num:12}

//订阅更新
store.subscribe(()=>{
   console.log(store.getState()); 
});

//派发事件
store.dispatch(action1);
store.dispatch(action2);
store.dispatch(action3);
store.dispatch(action4);
