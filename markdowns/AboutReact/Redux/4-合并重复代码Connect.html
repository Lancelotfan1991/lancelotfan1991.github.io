# 用Connect合并重复代码

1，在前面的案例中，about.js和home.js两个页面的代码中存在两个问题：

​ 第一：包含了大量重复的代码。

​ 第二：同时都对store产生了依赖（在实际开发中，我们希望对store产生依赖的地方越少越好）。

​ 因此，我们需要对代码进行重构，提取相同的部分。

​ 提取相同部分的关键是：

//每个组件跟redux产生关联的部分都是相同的：
//第一个是store，第二个是dispatch
//store中不同的是属性
//dispatch中不同的是参数和事件的类型

//因此，我们尝试构建一个connect.js，将组件和redux联系起来，

2，connect.js部分


import store from './store/index.js'

export function connect(mapStateToProps,mapDispatchToProps){
    return function enhanceHOC(WrappedComponent){
        constructor(props){
            super(props);
            this.state={
                storeState:this.store.getState();
            }
        }
        
        componentDidMount(){
            this.unsubscribe=this.store.subscribe(()=>{
                this.setState({
                    storeState:this.store.getState()
                })
            })
        }
        
        componentWillUnmount(){
            this.unsubscribe();
        }
        
        //将原本组件的state，store的state，store.dispatch都作为参数传入
        return <WrappedComponent
        		{...this.state}
        		{...mapStateToProps(this.store.getState())}
        		{...mapDispatchToProps(this.store.dispatch)}/>
    }
}

3，组件本身可以简化：

import React from 'react';

import {connect} from '../utitls/connect.js'

import { addAction ,subAction } from '../store/actionCreator.js'

//最终原本的属性，以及state和action都会作为属性传入该组件
const function About(props){
    return (
    	<div>
        	<h2>当前计数：props.counter </h2>
        	<button onClick={e => props.addAction()}>+5</button>
        	<button onClick={e => props.subAction()}>-5</button>
        </div>
    )
}

//从store中获取state，并最终以属性的形式传入组件
const mapStateToProps= state=>{
    return {
        counter:state.counter
    }
}

//从store上获取dispatch方法，最终触发事件时，都是调用的store.dispatch
const mapDispatchToProps= dispatch=>{
    return {
	    addAction:function(num){
           return dispatch(addAction(num));
        },
        subAction:function(num){
           return dispatch(subAction(num))
        }
	}
}

//将connect包装后的代码
export default connect(mapStateToProps,mapDispatchToProps)(About);
