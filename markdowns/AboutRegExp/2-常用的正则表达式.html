# 常用的正则表达式

# 1，验证是否是有效数字

/*
 *规则分析：
 *1.可能出现+,-也可能不出现  ^[+-]?
 *2.一位0-9都可以，多位首位不能为0	(\d|([1-9]\d+)) 要不然就是0-9，要不然就是1-9开头，然后重复多次0-9。
 *3.小数部分，可能有，可能没有，一旦有，后面必须有小数点加数字 （\.\d+）? 以小数点.开头，一段数字结尾，并重复0到1次
 */
 
reg=/^[+-]?(\d|([1-9]\d+))(\.\d+)?$/

# 2，验证密码

1，数字字母下划线
2，6-15位

let value=input.value;
let reg=/^\w{6-16}$/
let flag=reg.test(value);
flag==true && alert('符合密码规则');
flag==false && alert('不符合密码规则');

# 3，验证真实姓名

//1，尼古拉斯·赵四
//2，汉字的正则 
let reg=/^[\u4E00-\U9FA5]$/
//3，长度2-10位
//4，可能有译名	·汉字
let reg=/^[\u4E00]-\U9FA5]{2-10}(·[\u4E00-\u9FA5]{2,10}){0,2}$/

# 4，验证邮箱

let reg=/^\w+ ((-\w+)|(\.\w+))* @[A-Za-z0-9]+ ((\.|-)[A-Za-z0-9]+)* \.[A-Za-z0-9]+$/

//开头是数字字母下划线
//@符之前是数字字母下划线和.组成0到多次
//@
//@符之后是数字字母出现一到多位
//之后是-或者. 跟着数字字母 0到多次
//最后是.com .cn .org .net .edu这种，由点开始出现一到多次

举例：
lancelotfan@163.com.cn
zxt@zhufeng-peixun-office.com

# 5，身份证号码

//规则分析
//长度18位
//最后一位可能是X
由这两个规则导出的正则：
let reg=/^\d{17}(\d|X)$/

//身份证前六位：省市县 34 26 22
//中间八位：出生年月日
//最后四位： 最后一位：X|数字
		   倒数第二位：偶数女，奇数男
		   其余的是前面的14位，经过算法算出来的

//小括号第二个作用，分组捕获，不仅可以把大正则匹配的信息捕获到，还能捕获到每个小分组的内容
let reg=/^(\d{6})(\d{4})(\d{2})(\d{2})\d{2}(\d)(\d|X)$/
reg.exec('342622199107172956')
输出结果
["342622199107172956", "342622", "1991", "07", "17", "5", "6", index: 0, input: "342622199107172956", groups: undefined]
