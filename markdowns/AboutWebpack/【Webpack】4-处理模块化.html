# Webpack处理模块化

js的模块化一般意义上指的是两种：

一种是ES6推出的模块化，这种模块化，使用的是import和export字段来实现导入和导出。

另一种是commonjs推出的模块化，这种模块化，使用的是module.exports来进行导出，使用require来引入。

但是，尽管我们可以在编写代码时，使用这两种方式，浏览器事实上对于js模块化的支持是不完善的。

为了让浏览器能够正确识别模块化，我们就需要借助Webpack。

# 准备工作（如何配置）：

为了能看到webpack的打包后的代码能够分析，需要进行正确的设置：

const path = require('path');
const { CleanWebpackPlugin } = require('clean-webpack-plugin');
const HtmlWebpackPlugin = require('html-webpack-plugin');

module.exports = {
    mode: 'development',	// 核心1
    entry: './src/common_index.js',
    devtool: 'source-map',	// 核心2
    output: {
        filename: 'js/bundle.js',
        path: path.resolve(__dirname, './build')
    },
    plugins: [
        new HtmlWebpackPlugin({
            title: 'fantasy webpack'
        }),
        new CleanWebpackPlugin()
    ]
}

# 1，CommonJS实现原理

首先：我们需要写一个使用Commonjs实现模块化的js文件。

function dateFormat(date){
    return date;
}

function priceFormat(price){
    return price;
}

module.exports = {
    dateFormat,
    priceFormat
}

接着：我们在index.js中调用该模块。并输出。

const { dateFormat, priceFormat } = require('./js/format.js');

console.log(dateFormat('aaa'));
console.log(priceFormat('bbb'));

第三步：运行webpack打包：

我们可以得到编译后的bundle.js。

第四步：删去bundle.js的注释后，我们可以得到最终webpack帮助我们编译的代码：

var __webpack_modules__ = {
    // 模块的路径作为key
    // 模块的函数作为value
    "./src/js/format.js":
        ((module) => {
            function dateFormat(date) {
                return date;
            }
            function priceFormat(price) {
                return price;
            }
            module.exports = {
                dateFormat,
                priceFormat
            }
        })
};

var __webpack_module_cache__ = {};

function __webpack_require__(moduleId) {
    if (__webpack_module_cache__[moduleId]) {
        return __webpack_module_cache__[moduleId].exports;
    }
    var module = __webpack_module_cache__[moduleId] = {
        exports: {}
    };
    __webpack_modules__[moduleId](module, module.exports, __webpack_require__);
    return module.exports;
}

// 具体执行的代码
(() => {
    const { dateFormat, priceFormat } = __webpack_require__("./src/js/format.js");

    console.log(dateFormat('aaa'));
    console.log(priceFormat('bbb'));
})();

//# sourceMappingURL=bundle.js.map

这段代码的核心是：

（1）__webpack_modules__:存储了源代码中的所有模块，以路径为key。对象的内容是模块的映射。

（2）__webpack_require__方法：这个函数会从modules中去加载对应的模块。webpack将我们编写的代码中的require，替换为该方法。当我们调用这个方法时，便会从modules中调用对应模块的映射代码，并同时传入module，module.exports以及__webpack_require__函数本身。最终的返回值，便是原函数的module.exports。

（3）__webpack_module_cache__对象：用于缓存模块。这样第二次获取模块时，可以直接从缓存中获取指定的对象，而不会再次执行模块代码。比如第一次引用模块时，经过己算导出的是：{ a, b } 那么，第二次引用该模块时，便会直接得到a和b。

# 2，ES6 module实现原理

Webpack在处理es6的模块化代码时，与上面有所不同。区别在于：commonjs使用module.exports来集中的导出对象，而es的module，则可以导出多个对象。webpack需要将这些对象统一集中起来，形成一个模块。

首先我们需要写一个用esmodule：

export const sum =function(a, b){
    return a + b;
}

export const mul =function(a, b){
    return a * b;
}

然后在index.js中引用该模块：

import {sum, mul} from './js/math'

console.log('sum', sum(10, 20));
console.log('mul', mul(30, 40));

运行npm run webpack：

var __webpack_modules__ = ({
    "./src/esmodule_index.js":
        ((__unused_webpack_module, __webpack_exports__, __webpack_require__) => {

            __webpack_require__.r(__webpack_exports__);
            var _js_math__WEBPACK_IMPORTED_MODULE_0__ = __webpack_require__("./src/js/math.js");


            console.log('sum', (0, _js_math__WEBPACK_IMPORTED_MODULE_0__.sum)(10, 20));
            console.log('mul', (0, _js_math__WEBPACK_IMPORTED_MODULE_0__.mul)(30, 40));
        }),

    "./src/js/math.js":
        ((__unused_webpack_module, __webpack_exports__, __webpack_require__) => {
            
            // 调用了d函数，给exports设置了一个代理。
            // exports对象本身没有对应的函数。当我们调用sum时，事实上是调用的是原模块中的sum。
            __webpack_require__.r(__webpack_exports__);
            __webpack_require__.d(__webpack_exports__, {
                "sum": () => (sum),
                "mul": () => (mul)
            });
            const sum = function (a, b) {
                return a + b;
            }
            const mul = function (a, b) {
                return a * b;
            }
        })
});

var __webpack_module_cache__ = {};

function __webpack_require__(moduleId) {
    if (__webpack_module_cache__[moduleId]) {
        return __webpack_module_cache__[moduleId].exports;
    }
    var module = __webpack_module_cache__[moduleId] = {
        exports: {}
    };

    __webpack_modules__[moduleId](module, module.exports, __webpack_require__);

    return module.exports;
}

// 比较exports对象和definition对象的每一个key
// exports对象中不存在，而在definition中存在的key，会被复制到exports中
(() => {
    __webpack_require__.d = (exports, definition) => {
        for (var key in definition) {
            if (__webpack_require__.o(definition, key) && !__webpack_require__.o(exports, key)) {
                Object.defineProperty(exports, key, { enumerable: true, get: definition[key] });
            }
        }
    };
})();

// 判断一个obj是否拥有prop的属性
(() => {
    __webpack_require__.o = (obj, prop) => (Object.prototype.hasOwnProperty.call(obj, prop))
})();

// 如果可以使用symbol那么，就使用symbol，给exports对象添加一个Symbol.toStringTag为module
// 这样当我们对exports对象进行toString的时候，就可以得到[object Module]而不是[object object]
// 同时，我们也给exports对象添加一个__esModule属性，值为true，
// 这个r函数的作用，就是给一个对象添加module属性，告知这是一个模块对象。
(() => {
    __webpack_require__.r = (exports) => {
        if (typeof Symbol !== 'undefined' && Symbol.toStringTag) {
            Object.defineProperty(exports, Symbol.toStringTag, { value: 'Module' });
        }
        Object.defineProperty(exports, '__esModule', { value: true });
    };
})();

__webpack_require__("./src/esmodule_index.js");
